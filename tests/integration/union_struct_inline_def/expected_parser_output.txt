proc get_error_code(err: FileNotFound | Overflow | NoPermission | DivisionByZero) -> i32
{
    match err
    {
        case DivisionByZero()
        {
            return 1;
        }
        case Overflow()
        {
            return 2;
        }
        case FileNotFound()
        {
            return 3;
        }
        case NoPermission()
        {
            return 4;
        }
    }
}

proc main() -> i32
{
    let err: FileNotFound | Overflow | NoPermission | DivisionByZero = $InitUnion'(FileNotFound | Overflow | NoPermission | DivisionByZero)(FileNotFound());
    return get_error_code(err);
}

typealias Error = FileNotFound | Overflow | NoPermission | DivisionByZero;

typealias ArithError = Overflow | DivisionByZero;

typealias IOError = FileNotFound | NoPermission;

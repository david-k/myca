proc uninitialized'(T)() -> T
{
}

proc concat'(T, let count_a: usize, let count_b: usize)(a: ^[count_a]T, b: ^[count_b]T) -> [(count_a + count_b)]T
{
    let mut result: [(count_a + count_b)]T = uninitialized'([(count_a + count_b)]T)();
    let mut i: usize = 0;
    while (i < count_a)
    {
        ((result)^ = ((a)^)^);
        (i = (i + 1));
    }
    while (i < (count_a + count_b))
    {
        ((result)^ = ((b)^)^);
        (i = (i + 1));
    }
    return result;
}

proc array_sum'(let N: usize)(arr: [N]i32) -> i32
{
    let mut sum: i32 = 0;
    let mut i: usize = 0;
    while (i < N)
    {
        (sum = (sum + (arr)^));
        (i = (i + 1));
    }
    return sum;
}

proc main() -> i32
{
    let mut a: [3]i32 = uninitialized'([3]i32)();
    ((a)^ = 7);
    ((a)^ = 11);
    ((a)^ = 13);
    let mut b: [2]i32 = uninitialized'([2]i32)();
    ((b)^ = 23);
    ((b)^ = 42);
    return array_sum'(5)(concat'(i32, 3, 2)(&(a), &(b)));
}
